CSS-код елемента: 

.pagination a {
text-decoration: none;
color: #2d2119;
font-size: 16px;
font-weight: 400;
position: relative;
}

/* Роздільні лінії */
.pagination a:not(:last-child)::after {
content: "";
position: absolute;
right: -55px;
/* відстань від цифри до лінії */
top: 50%;
transform: translateY(-50%);
width: 30px;
height: 1px;
background: #111;
}

/* Активна цифра */
.pagination a.active {
font-size: 22px;
/* більша цифра */

}
