Завдання:

Напишіть функцію, яка приймає массив об'єктів і повертає новий массив об'єктів,у якому є знижка на 30 % на всі медикаменти ціна, яких перевищує 300 грн у масиві Надайте ід для кожного медикамента const medicines = [ { name: "Noshpa", price: 170 }, { name: "Analgin", price: 55 }, { name: "Quanil", price: 310 }, { name: "Alphacholine", price: 390 }, ];

Код:

                                                    
    
    const medicines = [
    { name: "Noshpa", price: 170 },
    { name: "Analgin", price: 55 },
    { name: "Quanil", price: 310 },
    { name: "Alphacholine", price: 390 },
    ];
    
    function getMedsWithDiscountAndId(medsArray) {
    // Використовуємо .map() для створення нового масиву
    return medsArray.map((medicine, index) => {
    
    // Створюємо копію об'єкта, щоб не змінювати вихідний масив
    const updatedMedicine = { ...medicine };
    
    // Надаємо ID
    // crypto.randomUUID() - надійний спосіб, але для простоти можна і так:
    updatedMedicine.id = `med-${index + 1}`;
    
    // Застосовуємо знижку 30% (тобто ціна = 70% від старої)
    if (updatedMedicine.price > 300) {
    // .toFixed(2) - для округлення до 2 знаків після коми
    updatedMedicine.price = Number((updatedMedicine.price * 0.7).toFixed(2));
    }
    
    return updatedMedicine;
    });
    }
    
    // --- Перевірка ---
    console.log('\n--- Завдання 1.6 ---');
    const updatedMedicines = getMedsWithDiscountAndId(medicines);
    console.log('Медикаменти зі знижками та ID:', updatedMedicines);
    console.log('Оригінальний масив (незмінений):', medicines);
