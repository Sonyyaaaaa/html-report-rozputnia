Завдання:

Напиши функцію конструктор Storage який створює об'єкти Для управління складом товарів. При виклику отримуватиме один агрумент - початковий масив товарів, і записувати їх у властивість items. Додай методи класу: getItems() - повертайте масив товарів addItems(item) - отримує новий товар та додає його до поточних removeItem(item) - отримує товар і, якщо він є, видаляє його з поточних const arr = ["apple", "banana", "mango"]

Код:

                                                    

function Storage(initialItems) {
this.items = initialItems || []; // Початковий масив
}

// Метод getItems()
Storage.prototype.getItems = function() {
return this.items;
};

// Метод addItems(item)
Storage.prototype.addItems = function(item) {
this.items.push(item);
console.log(`Товар "${item}" додано.`);
};

// Метод removeItem(item)
Storage.prototype.removeItem = function(item) {
const itemIndex = this.items.indexOf(item);

if (itemIndex !== -1) {
this.items.splice(itemIndex, 1);
console.log(`Товар "${item}" видалено.`);
} else {
console.warn(`Товар "${item}" на складі не знайдено.`);
}
};


// --- Перевірка ---
console.log('\n--- Завдання 1.8 ---');
const arr = ["apple", "banana", "mango"];
const storage = new Storage(arr);

console.log('Початковий склад:', storage.getItems());
storage.addItems('orange');
console.log('Після додавання:', storage.getItems());
storage.removeItem('banana');
console.log('Після видалення:', storage.getItems());
storage.removeItem('grape'); // Перевірка видалення неіснуючого
