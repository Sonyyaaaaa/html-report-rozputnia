Завдання:

Напишіть функцію checkBrackets(str) яка приймає рядок жс коду (someFn) і перевіряє правильність закриття дужок () {} [] Якщо рядок містить коректний код функція повертає true. В іншому випадку повертає false

Код:

                                                    


function checkBrackets(str) {
const stack = []; // Стек для зберігання відкриваючих дужок

// Карта відповідності дужок
const bracketsMap = {
'(': ')',
'[': ']',
'{': '}'
};

const openBrackets = Object.keys(bracketsMap);
const closeBrackets = Object.values(bracketsMap);

for (const char of str) {
if (openBrackets.includes(char)) {
// Якщо це відкриваюча дужка - додаємо її в стек
stack.push(char);
} else if (closeBrackets.includes(char)) {
// Якщо це закриваюча дужка
if (stack.length === 0) {
return false; // Немає відповідної відкриваючої
}

const lastOpenBracket = stack.pop(); // Беремо останню відкриваючу

// Перевіряємо, чи відповідає вона поточній закриваючій
if (bracketsMap[lastOpenBracket] !== char) {
return false; // Неправильний тип дужки (напр. (])
}
}
// Всі інші символи ігноруємо
}

// Якщо в кінці стек порожній - всі дужки закрито правильно
return stack.length === 0;
}
