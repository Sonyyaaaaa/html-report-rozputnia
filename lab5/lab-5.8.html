Застосувати функції 2.Упорядкувати масив у порядку зростання методом бульбашки. Надрукувати вхідний та вихідний масив. Застосувати функції

/**
* Створює масив, запитуючи кожен елемент у користувача.
* @param {number} size - Кількість елементів у масиві.
* @param {string} arrayName - Назва масиву для prompt.
* @returns {number[]} - Створений масив.
*/
function createArrayFromUserInput(size, arrayName) {
const arr = [];
for (let i = 0; i < size; i++) { let value=parseFloat(prompt(`Введіть елемент ${i + 1}/${size} для масиву
    ${arrayName}:`)); // Перевірка на випадок, якщо користувач ввів не число if (isNaN(value)) { alert("Ви ввели не
    число. Елементу буде присвоєно значення 0."); value=0; } arr.push(value); } return arr; } /** * Знаходить спільні
    елементи між двома масивами. * @param {number[]} arr1 - Перший масив. * @param {number[]} arr2 - Другий масив. *
    @returns {number[]} - Масив унікальних спільних елементів. */ function findCommonElements(arr1, arr2) { const
    common=[]; for (const item of arr1) { // Якщо елемент є в arr2 і його ще немає в масиві common if
    (arr2.includes(item) && !common.includes(item)) { common.push(item); } } return common; } /** * Знаходить
    мінімальний елемент у масиві. * @param {number[]} arr - Масив чисел. * @returns {number | null} - Мінімальне число
    або null, якщо масив порожній. */ function findMinElement(arr) { if (arr.length===0) { return null; // Немає
    елементів для пошуку } // Використовуємо Math.min з оператором spread (...) return Math.min(...arr); } /** * Сортує
    масив методом "бульбашки" . * @param {number[]} arr - Вхідний масив. * @returns {number[]} - Новий, відсортований
    масив. */ function bubbleSort(arr) { // Створюємо копію, щоб не змінювати оригінальний масив const
    sortedArr=[...arr]; const n=sortedArr.length; for (let i=0; i < n - 1; i++) { let swapped=false; // Оптимізація for
    (let j=0; j < n - 1 - i; j++) { // Якщо поточний елемент більший за наступний if (sortedArr[j]> sortedArr[j + 1]) {
    // Міняємо їх місцями
    [sortedArr[j], sortedArr[j + 1]] = [sortedArr[j + 1], sortedArr[j]];
    swapped = true;
    }
    }
    // Якщо за весь внутрішній цикл не було обмінів, масив відсортовано
    if (!swapped) {
    break;
    }
    }
    return sortedArr;
    }


    // --- ГОЛОВНА ЛОГІКА ВИКОНАННЯ ---

    // === Виконання Завдання 7.1 ===
    console.log("--- ЗАВДАННЯ 7.1 ---");
    console.log("УМОВА: Знайти найменший спільний елемент у двох векторах.");

    // 1. Створюємо масиви
    let sizeA = parseInt(prompt("Завдання 7.1: Введіть розмір ПЕРШОГО масиву (A):"), 10);
    const arrayA = createArrayFromUserInput(sizeA, "A");

    let sizeB = parseInt(prompt("Завдання 7.1: Введіть розмір ДРУГОГО масиву (B):"), 10);
    const arrayB = createArrayFromUserInput(sizeB, "B");

    console.log("Перший масив (A):", arrayA);
    console.log("Другий масив (B):", arrayB);

    // 2. Знаходимо спільні елементи
    const commonElements = findCommonElements(arrayA, arrayB);

    if (commonElements.length > 0) {
    console.log("Спільні елементи:", commonElements);

    // 3. Знаходимо мінімальний серед спільних
    const minCommon = findMinElement(commonElements);
    console.log(`Найменший серед спільних елементів: ${minCommon}`);
    } else {
    console.log("Спільних елементів не знайдено.");
    }
