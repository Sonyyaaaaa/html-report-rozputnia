JS:

                            
                            
                            
                            // Отримуємо доступ до елементів сторінки
                            const timeInput = document.getElementById('alarmTimeInput');
                            const setButton = document.getElementById('setAlarm');
                            const cancelButton = document.getElementById('cancelAlarm');
                            const statusDiv = document.getElementById('status');
                            
                            // Змінна для зберігання ID таймера
                            let alarmTimerId = null;
                            
                            /**
                             * Функція, яка спрацює, коли прийде час
                             */
                            function triggerAlarm() {
                                console.log("БУДИЛЬНИК СПРАЦЮВАВ!");
                                statusDiv.textContent = "⏰ ДЗИНЬ-ДЗИНЬ! ⏰";
                                alert("⏰ ДЗИНЬ-ДЗИНЬ! Час прокидатись!");
                                alarmTimerId = null;
                            }
                            
                            /**
                             * Логіка кнопки "Встановити"
                             */
                            setButton.addEventListener('click', () => {
                                // 1. Отримуємо значення з поля вводу
                                const alarmTimeValue = timeInput.value;
                            
                                // Перевірка, чи щось введено
                                if (!alarmTimeValue) {
                                    alert("Будь ласка, оберіть дату та час.");
                                    return;
                                }
                            
                                // 2. Конвертуємо рядок в об'єкт Date
                                const alarmTime = new Date(alarmTimeValue);
                                const now = new Date();
                            
                                // 3. Розраховуємо час до спрацювання (у мілісекундах)
                                const timeToAlarm = alarmTime.getTime() - now.getTime();
                            
                                // 4. Перевіряємо, чи час у майбутньому
                                if (timeToAlarm <= 0) {
                                    alert("Будь ласка, оберіть час у майбутньому.");
                                    return;
                                }
                            
                                // Якщо вже є інший будильник - скасуємо його
                                if (alarmTimerId) {
                                    clearTimeout(alarmTimerId);
                                }
                            
                                // 5. Встановлюємо таймер
                                // setTimeout викликає функцію triggerAlarm через 'timeToAlarm' мілісекунд
                                alarmTimerId = setTimeout(triggerAlarm, timeToAlarm);
                            
                                // 6. Повідомляємо користувача
                                const alarmDateStr = alarmTime.toLocaleString('uk-UA');
                                console.log(`Будильник встановлено на ${alarmDateStr}`);
                                statusDiv.textContent = `Будильник встановлено на ${alarmDateStr}`;
                            });
                            
                            /**
                             * Логіка кнопки "Вимкнути"
                             */
                            cancelButton.addEventListener('click', () => {
                                if (alarmTimerId) {
                                    // Скасовуємо таймер, який був встановлений
                                    clearTimeout(alarmTimerId);
                                    alarmTimerId = null;
                                    
                                    console.log("Будильник скасовано.");
                                    statusDiv.textContent = "Будильник скасовано.";
                                } else {
                                    console.log("Немає активних будильників.");
                                    statusDiv.textContent = "Немає активних будильників.";
                                }
                            });
